## 季节性

ggseasonplot(a10, year.labels=TRUE, year.labels.left=TRUE) +
  xlab("月份")+
  ylab("百万（美元）") +
  ggtitle("季节图：降糖药物销量")+
  theme(text = element_text(family = "STHeiti"))+
  theme(plot.title = element_text(hjust = 0.5)) +
  theme_clean()

ggseasonplot(a10, polar=TRUE) +
  xlab("月份")+
  ylab("百万（美元）") +
  ggtitle("极坐标季节图：降糖药物销量")+
  theme(text = element_text(family = "STHeiti"))+
  theme(plot.title = element_text(hjust = 0.5))+
  theme_clean()
