import torch # 创建张量 x = torch.tensor([[1.0, 2.0], [3.0, 4.0]]) # 从列表创建 x = torch.zeros(3, 3) # 全零 x = torch.randn(2, 3) # 标准正态分布 # 形状与设备 x.shape # torch.Size([2, 3]) x.to('cuda') # 搬到 GPU（如果可用） # 自动求导：requires_grad=True 表示需要计算梯度 x = torch.tensor(2.0, requires_grad=True) y = x ** 2 y.backward() # 反向传播 x.grad # 梯度值：4.0

定义神经网络

import torch.nn as nn import torch.nn.functional as F class Net(nn.Module): def __init__(self): super().__init__() self.fc1 = nn.Linear(784, 128) # 输入 784，隐藏层 128 self.fc2 = nn.Linear(128, 10) # 输出 10 类 def forward(self, x): x = x.view(-1, 784) # 展平 x = F.relu(self.fc1(x)) # 激活函数 x = self.fc2(x) return x model = Net() optimizer = torch.optim.SGD(model.parameters(), lr=0.01) criterion = nn.CrossEntropyLoss() # 分类任务常用损失 # 训练一步 for data, target in dataloader: optimizer.zero_grad() # 清空梯度 output = model(data) loss = criterion(output, target) loss.backward() # 反向传播 optimizer.step() # 更新参数
